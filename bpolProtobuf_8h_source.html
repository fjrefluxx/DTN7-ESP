29void decodeProtobuf(uint8_t* data, size_t datasize) {
30 ESP_LOGI("decode proto", "protoVersion %s", protobuf_c_version());
31
32 // create overall packet Object from received data
33 Lora__Protocol__Packet* packet =
34 lora__protocol__packet__unpack(NULL, datasize, data);
35
36 // check if decoding was successful
37 if (!packet) {
38 ESP_LOGE("decode proto", "failed");
39 return;
40 }
41
42 // handle the different types of BPoL packets
43 switch (packet->type) {
44 case LORA__PROTOCOL__PACKET_TYPE__TYPE_BUNDLE_FORWARD: {
45 // if packet is of type BundleForward, the contained Bundle must be extracted and send to the receiveQueue
46
47 // first decode the bundle from its CBOR representation
48 Bundle* received =
49 Bundle::fromCbor(packet->bundle_forward->bundle_data.data,
50 packet->bundle_forward->bundle_data.len);
51
52 // check the Bundles validity
53 if (received->valid) {
54 // attempt to get the sending node from storage. If it is already known the corresponding node object is returned, otherwise one is created
55 Node sender = DTN7::BPA->storage->getNode(
56 std::string(packet->bundle_forward->sender));
57
58 // if the URI of the created or stored node is empty, update it to the now known URI
59 if (sender.URI == "none")
60 sender.URI = std::string(packet->bundle_forward->sender);
61
62 // set the last seen time for the sending node
63 sender.setLastSeen();
64
65 // store the sender node in the list of known nodes
66 DTN7::BPA->storage->addNode(sender);
67
68 // create a received bundle containing the bundle and the URI of the sender
69 ReceivedBundle* recBundle =
70 new ReceivedBundle(received, sender.URI);
71
72 // send the Received bundle to the receiveQueue for further processing
73 xQueueSend(DTN7::BPA->receiveQueue, (void*)&recBundle,
74 portMAX_DELAY);
75 }
76 else {
77 delete received; // invalid bundles are discarded directly
78 ESP_LOGW("LoraCLARecTask", "deleted Invalid Bundle");
79 }
80 break;
81 }
82 case LORA__PROTOCOL__PACKET_TYPE__TYPE_ADVERTISE: {
83 // if packet is of type Advertise, the contained information about the node has to be updated/ added to the storage
84 ESP_LOGW("LoraCLARecTask", "received advertise");
85
86 // attempt to get the sending node from storage. If it is already known the corresponding node object is returned, otherwise one is created
87 Node sender = DTN7::BPA->storage->getNode(
88 std::string(packet->advertise->node_name));
89 // if the URI of the created or stored node is empty, update it to the now known URI
90 if (sender.URI == "none")
91 sender.URI = std::string(packet->advertise->node_name);
92
93#if CONFIG_useReceivedSet
94 // if usage of hashes of bundleIDs for reception confirmation is enabled in menuconfig, these hashes have to be read from the data
95 // field of the advertise message and inserted in the set of hashes received from this sender
96
97 // first find the corresponding entry in the data field
98 for (int i = 0; i < packet->advertise->n_data; i++) {
99 if (strcmp(packet->advertise->data[i]->key, "BH") == 0) {
100 // read bundlesHash from data map, this is a sting oh hashes delimited by ';'
101 std::string bundlesHashes =
102 std::string(packet->advertise->data[i]->value);
103
104 // now split the string into the individual hashes for this a string stream is generated and split at the delimiter ';'
105 std::stringstream hashesStream(bundlesHashes);
106 std::string hashString;
107 while (std::getline(hashesStream, hashString, ';')) {
108 size_t hash = 0;
109 std::stoul(hashString.c_str(), &hash);
110 // add each hash to the received Hashes
111 sender.receivedHashes.insert(hash);
112 }
113 }
114 }
115#endif
116 // set the last seen time for the sending node
117 sender.setLastSeen();
118
119 // ToDo Position: The advertise message can potentially contain position information. This would have to be decoded here
120
121 // store the sender node in the list of known nodes
122 DTN7::BPA->storage->addNode(sender);
123 } break;
124 default:
125 break;
126 }
127
128 // cleanup packet
129 lora__protocol__packet__free_unpacked(packet, NULL);
130 return;
131}
