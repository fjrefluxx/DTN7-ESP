207 uint64_t insertCanonicalBlock(CanonicalBlock block) {
208 uint64_t newNum = block.blockNumber;
209 PayloadBlock oldPayload = payloadBlock;
210 // check if primary block
211 if (block.blockTypeCode == 1) {
212 if (this->payloadBlock.dataSize == 0) {
213 // if the current payload is empty, add given block as payload
214 this->payloadBlock = PayloadBlock(block);
215 ESP_LOGE("Bundle insert cBlock", "Payload set!");
216 }
217 else {
218 ESP_LOGE("Bundle insert cBlock",
219 "Bundle already has payload, if it should be "
220 "overriden use setPayload()");
221 }
222 }
223 else {
224 if (newNum == 0 ||
225 usedBlockNums.find(newNum) != usedBlockNums.end()) {
226 // if the given block has number 0 or a number which is already in use in the bundle, it needs to be given a new number
227 newNum =
228 2; // lowest block number for a canonical block can be 2, because primary block is always 1
229 while (usedBlockNums.find(newNum) != usedBlockNums.end()) {
230 newNum += 1;
231 }
232 }
233 block.blockNumber = newNum;
234 extensionBlocks.push_back(block);
235 usedBlockNums.insert(newNum);
236 if (block.blockTypeCode == 6)
237 hasPreviousNode = true;
238 if (block.blockTypeCode == 7)
239 hasBundleAge = true;
240 if (block.blockTypeCode == 10)
241 hasHopCount = true;
242 }
243 return newNum;
244 }

249 CanonicalBlock removeBlock(uint64_t blockNumber) {
