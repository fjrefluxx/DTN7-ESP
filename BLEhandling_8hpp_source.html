152 struct ble_gatt_access_ctxt* ctxt, void* arg) {
153 if (!(ctxt->op == BLE_GATT_ACCESS_OP_WRITE_CHR))

156 uint8_t recdata[ctxt->om->om_len];
157 ble_hs_mbuf_to_flat(ctxt->om, recdata, ctxt->om->om_len, NULL);
158 /* uncomment to print received data
159 for (size_t i = 0; i < sizeof(recdata); i++) {
160 printf("%02X ", recdata[i]); // %02X prints each byte in two-digit hex format
161 }
162 printf("\n");
163 */
164 ble_gap_terminate(
165 conn_handle,
166 BLE_ERR_REM_USER_CONN_TERM); // terminate the connection upon reception of the data
167
168 if (recdata[0] == 0x9f) {
169 struct ble_gap_conn_desc desc;
170 int rc = ble_gap_conn_find(conn_handle, &desc);
171 if (rc == 0) {
172 // here could be filtered by rssi
173 int8_t rssi;
174 ble_gap_conn_rssi(conn_handle, &rssi);
175 ESP_LOGI("BLE Receiver",

177 conn_handle, rssi);
178 Bundle* received = Bundle::fromCbor(recdata, sizeof(recdata));
179 handleBLEReception(received, &desc.peer_id_addr);
180 }
181 }
