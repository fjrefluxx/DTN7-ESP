142static int ble_write_callback(uint16_t conn_handle, uint16_t attr_handle,
143 struct ble_gatt_access_ctxt* ctxt, void* arg) {
144 if (!(ctxt->op == BLE_GATT_ACCESS_OP_WRITE_CHR))
145 return 0;
146 ESP_LOGI("BLE Receiver", "Received data with length: %u", ctxt->om->om_len);
147 uint8_t recdata[ctxt->om->om_len];
148 ble_hs_mbuf_to_flat(ctxt->om, recdata, ctxt->om->om_len, NULL);
149 /* uncomment to print received data
150 for (size_t i = 0; i < sizeof(recdata); i++) {
151 printf("%02X ", recdata[i]); // %02X prints each byte in two-digit hex format
152 }
153 printf("\n");
154 */
155 ble_gap_terminate(
156 conn_handle,
157 BLE_ERR_REM_USER_CONN_TERM); // terminate the connection upon reception of the data
158
159 if (recdata[0] == 0x9f) {
160 struct ble_gap_conn_desc desc;
161 int rc = ble_gap_conn_find(conn_handle, &desc);
162 if (rc == 0) {
163 // here could be filtered by rssi
164 int8_t rssi;
165 ble_gap_conn_rssi(conn_handle, &rssi);
166 ESP_LOGI("BLE Receiver",
167 "Received data from conn_handle = %d (RSSI = %d dBm)",
168 conn_handle, rssi);
169 Bundle* received = Bundle::fromCbor(recdata, sizeof(recdata));
170 handleBLEReception(received, &desc.peer_id_addr);
171 }
172 }
173 // printf("recdata[0]:%u ,recdata[sizeof(recdata-3)]:%u\n" ,recdata[0], recdata[sizeof(recdata-3)]);
174 return 0;
175}
